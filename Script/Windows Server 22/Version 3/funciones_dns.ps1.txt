# Cargar funciones comunes si no están disponibles
if (-not (Get-Command "Write-Color" -ErrorAction SilentlyContinue)) {
    . "$PSScriptRoot\funciones_compartidas.ps1"
}

function Verificar-InstalacionDNS {
    Write-Color "`n--- VERIFICANDO ROL DNS (WINDOWS SERVER) ---" $CYAN
    
    $dnsServer = Get-WindowsFeature -Name DNS
    
    if ($dnsServer.Installed) {
        Write-Color "[OK] DNS Server instalado" $GREEN
        
        $dnsService = Get-Service -Name DNS -ErrorAction SilentlyContinue
        if ($dnsService -and $dnsService.Status -eq "Running") {
            Write-Color "      ✓ Servicio DNS activo" $GREEN
        } else {
            Write-Color "      ✗ Servicio DNS inactivo" $RED
        }
    } else {
        Write-Color "[X] DNS Server NO instalado" $RED
    }
}

function Instalar-DNS {
    Write-Color "Instalando servidor DNS..." $YELLOW
    
    try {
        Install-WindowsFeature -Name DNS -IncludeManagementTools
        
        if ($?) {
            Write-Color "✓ Instalación DNS completada" $GREEN
            
            # Configurar servicio para inicio automático
            Set-Service -Name DNS -StartupType Automatic
            Start-Service -Name DNS
            Write-Color "✓ Servicio DNS iniciado y configurado" $GREEN
        }
    } catch {
        Write-Color "✗ Error en la instalación DNS: $_" $RED
    }
}

function Reiniciar-ServicioDNS {
    Write-Color "Reiniciando servicio DNS..." $YELLOW
    
    try {
        Restart-Service -Name DNS -Force
        Write-Color "✓ Servicio DNS reiniciado" $GREEN
    } catch {
        Write-Color "✗ Error al reiniciar DNS: $_" $RED
    }
}

function Agregar-ZonaDNS {
    $SERVER_IP = Obtener-IPActual
    if (-not $SERVER_IP) {
        $SERVER_IP = "10.0.0.1"
    }
    
    Write-Color "`n--- AGREGAR ZONA DNS (CON IP VIRTUAL AUTOMÁTICA) ---" $YELLOW
    Write-Color "NOTA: Se creará automáticamente una IP virtual en el servidor" $CYAN
    
    $dominio = Read-Host "Nombre del Dominio (ej. laboratorio.local)"
    if ([string]::IsNullOrEmpty($dominio)) { return }
    
    do {
        $virtualIP = Read-Host "IP virtual para $dominio (ej. 10.0.0.6)"
    } while (-not (Validar-IPCompleta $virtualIP))
    
    if ($virtualIP -eq $SERVER_IP) {
        Write-Color "Error: No puedes usar la IP del servidor ($SERVER_IP) como IP virtual" $RED
        return
    }
    
    Write-Color "`nPASO 1: Creando IP virtual" $CYAN
    if (-not (Crear-IPVirtual $virtualIP $INTERFACE)) {
        Write-Color "ERROR CRÍTICO: No se pudo crear la IP virtual. Abortando." $RED
        return
    }
    
    Write-Color "`nPASO 2: Configurando zona DNS" $CYAN
    
    try {
        # Crear zona directa
        Add-DnsServerPrimaryZone -Name $dominio -ZoneFile "$dominio.dns" -DynamicUpdate None
        
        # Agregar registros
        Add-DnsServerResourceRecordA -Name "@" -ZoneName $dominio -IPv4Address $virtualIP -CreatePtr
        Add-DnsServerResourceRecordA -Name "www" -ZoneName $dominio -IPv4Address $virtualIP -CreatePtr
        Add-DnsServerResourceRecordA -Name "mail" -ZoneName $dominio -IPv4Address $virtualIP -CreatePtr
        Add-DnsServerResourceRecordA -Name "ftp" -ZoneName $dominio -IPv4Address $virtualIP -CreatePtr
        
        # Agregar servidores de nombres
        Add-DnsServerResourceRecordA -Name "ns1" -ZoneName $dominio -IPv4Address $SERVER_IP
        Add-DnsServerResourceRecordA -Name "ns2" -ZoneName $dominio -IPv4Address $SERVER_IP
        
        # Agregar registro MX
        Add-DnsServerResourceRecordMX -Name "." -ZoneName $dominio -MailExchange "mail.$dominio" -Preference 10
        
        Write-Color "✓ Zona DNS creada exitosamente" $GREEN
    }
    catch {
        Write-Color "✗ Error al crear zona DNS: $_" $RED
        return
    }
    
    Write-Color "`nPASO 3: Reiniciando servicio DNS" $CYAN
    Reiniciar-ServicioDNS
    
    Write-Color "`n✅ CONFIGURACIÓN COMPLETADA EXITOSAMENTE" $GREEN
    Write-Color "═══════════════════════════════════════════" $CYAN
    Write-Color "  Dominio:        $dominio" $GREEN
    Write-Color "  IP Virtual:     $virtualIP" $GREEN
    Write-Color "  Servidor DNS:   $SERVER_IP" $GREEN
    Write-Color "  Interfaz:       $INTERFACE" $GREEN
    Write-Color "═══════════════════════════════════════════" $CYAN
    
    Write-Color "`nPRUEBAS DESDE CLIENTES:" $YELLOW
    Write-Color "  ping $virtualIP      # ✅ Debe funcionar" $WHITE
    Write-Color "  ping $dominio         # ✅ Debe funcionar" $WHITE
    Write-Color "  nslookup $dominio     # ✅ Debe mostrar $virtualIP" $WHITE
}

function Eliminar-ZonaDNS {
    Write-Color "`n--- ELIMINAR ZONA DNS ---" $YELLOW
    
    $zonas = Get-DnsServerZone | Where-Object { $_.ZoneName -notin "RootHints", "..msdcs", "TrustAnchors" }
    
    if ($zonas) {
        Write-Color "Zonas actuales:" $CYAN
        $zonas | ForEach-Object { Write-Host "  - $($_.ZoneName)" }
    } else {
        Write-Color "No hay zonas configuradas." $YELLOW
        return
    }
    
    $zonaDel = Read-Host "Nombre EXACTO de la zona a borrar"
    if ([string]::IsNullOrEmpty($zonaDel)) { return }
    
    $zona = Get-DnsServerZone -Name $zonaDel -ErrorAction SilentlyContinue
    if ($zona) {
        # Buscar IP virtual asociada
        $records = Get-DnsServerResourceRecord -ZoneName $zonaDel -RRType A
        $virtualIP = ($records | Where-Object { $_.HostName -eq "@" }).RecordData.IPv4Address
        
        if ($virtualIP) {
            Write-Color "Eliminando IP virtual asociada: $virtualIP" $CYAN
            Eliminar-IPVirtual $virtualIP $INTERFACE
        }
        
        Remove-DnsServerZone -Name $zonaDel -Force
        Write-Color "✓ Zona eliminada" $GREEN
    } else {
        Write-Color "✗ Zona no encontrada" $RED
    }
}

function Listar-Zonas {
    Write-Color "`n--- DOMINIOS CONFIGURADOS ---" $CYAN
    
    $zonas = Get-DnsServerZone | Where-Object { $_.ZoneName -notin "RootHints", "..msdcs", "TrustAnchors" }
    
    if ($zonas) {
        foreach ($zona in $zonas) {
            $records = Get-DnsServerResourceRecord -ZoneName $zona.ZoneName -RRType A -ErrorAction SilentlyContinue
            $ipVirtual = ($records | Where-Object { $_.HostName -eq "@" }).RecordData.IPv4Address
            Write-Host "  - $($zona.ZoneName) → $ipVirtual (IP virtual)"
        }
    } else {
        Write-Color "No hay zonas configuradas." $YELLOW
    }
    
    Write-Color "`nIPs activas en $INTERFACE:" $CYAN
    Get-NetIPAddress -InterfaceAlias $INTERFACE -AddressFamily IPv4 | Format-Table IPAddress
}

function submenu_dns {
    while ($true) {
        Clear-Host
        Write-Color "`n=== GESTIÓN DE DOMINIOS DNS (CON IPS VIRTUALES) ===" $CYAN
        Write-Host "1) Agregar Dominio (crea IP virtual automáticamente)"
        Write-Host "2) Eliminar Dominio (elimina IP virtual asociada)"
        Write-Host "3) Ver Dominios e IPs Virtuales"
        Write-Host "4) Volver al Menú Principal"
        
        $subopc = Read-Host "`nSeleccione opción"
        
        switch ($subopc) {
            "1" { Agregar-ZonaDNS; Read-Host "Presiona Enter..." }
            "2" { Eliminar-ZonaDNS; Read-Host "Presiona Enter..." }
            "3" { Listar-Zonas; Read-Host "Presiona Enter..." }
            "4" { return }
            default { Write-Color "Opción inválida" $RED; Start-Sleep -Seconds 1 }
        }
    }
}
