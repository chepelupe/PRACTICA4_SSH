# ==============================================================================
# FUNCIONES COMUNES - Para Windows Server 2022 (PowerShell)
# ==============================================================================

# COLORES (para PowerShell)
$RED = "Red"
$GREEN = "Green"
$YELLOW = "Yellow"
$CYAN = "Cyan"
$WHITE = "White"

# =========================
# VARIABLES GLOBALES
# =========================
$INTERFACE = ""
$IPS_VIRTUALES = @()

# =========================
# FUNCIONES DE VALIDACIÓN
# =========================

function Write-Color {
    param(
        [string]$Message,
        [string]$Color = "White"
    )
    Write-Host $Message -ForegroundColor $Color
}

function Check-Administrator {
    $currentPrincipal = New-Object Security.Principal.WindowsPrincipal([Security.Principal.WindowsIdentity]::GetCurrent())
    if (-not $currentPrincipal.IsInRole([Security.Principal.WindowsBuiltInRole]::Administrator)) {
        Write-Color "Error: Este script debe ejecutarse como ADMINISTRADOR" $RED
        Write-Color "Por favor, ejecuta PowerShell como Administrador" $YELLOW
        Read-Host "Presiona Enter para salir"
        exit 1
    }
    Write-Color "✓ Ejecutando como Administrador" $GREEN
}

function Validar-IPSintaxis {
    param([string]$ip)
    
    if ($ip -match '^([0-9]{1,3}\.){3}[0-9]{1,3}$') {
        $octetos = $ip.Split('.')
        foreach ($octeto in $octetos) {
            if ([int]$octeto -lt 0 -or [int]$octeto -gt 255) {
                return $false
            }
        }
        return $true
    }
    return $false
}

function Es-IPProhibida {
    param([string]$ip)
    
    if ($ip -eq "0.0.0.0" -or $ip -eq "255.255.255.255") { return $true }
    if ($ip -like "127.*") { return $true }
    
    $firstOctet = [int]($ip.Split('.')[0])
    if ($firstOctet -ge 224) { return $true }
    
    return $false
}

function Validar-IPCompleta {
    param([string]$ip)
    
    if (Validar-IPSintaxis $ip) {
        if (-not (Es-IPProhibida $ip)) {
            return $true
        }
    }
    return $false
}

# =========================
# FUNCIONES DE RED
# =========================

function Seleccionar-Interfaz {
    Clear-Host
    Write-Color "--- SELECCIÓN DE INTERFAZ DE RED ---" $CYAN
    
    $interfaces = Get-NetAdapter | Where-Object { $_.Status -eq "Up" }
    
    Write-Color "`nInterfaces disponibles:" $YELLOW
    $index = 1
    $interfaceList = @{}
    
    foreach ($iface in $interfaces) {
        $ipInfo = Get-NetIPAddress -InterfaceIndex $iface.ifIndex -AddressFamily IPv4 -ErrorAction SilentlyContinue
        $ipAddress = if ($ipInfo) { $ipInfo.IPAddress } else { "Sin IP" }
        Write-Host "  $index) $($iface.Name) - $($iface.InterfaceDescription) - IP: $ipAddress"
        $interfaceList[$index] = $iface.Name
        $index++
    }
    
    do {
        $selection = Read-Host "`nSelecciona el número de la interfaz"
    } while (-not $interfaceList.ContainsKey([int]$selection))
    
    $INTERFACE = $interfaceList[[int]$selection]
    Write-Color "`nTrabajando sobre: $INTERFACE" $GREEN
    Start-Sleep -Seconds 1
}

function Obtener-IPActual {
    $ipInfo = Get-NetIPAddress -InterfaceAlias $INTERFACE -AddressFamily IPv4 -ErrorAction SilentlyContinue
    if ($ipInfo) {
        return $ipInfo.IPAddress
    }
    return $null
}

function Configurar-FirewallPing {
    Write-Color "Configurando Firewall para permitir PING (ICMP)..." $CYAN
    
    # Habilitar reglas de ICMPv4
    New-NetFirewallRule -DisplayName "ICMP Allow (PING)" -Direction Inbound -Protocol ICMPv4 -Action Allow -ErrorAction SilentlyContinue
    New-NetFirewallRule -DisplayName "ICMP Allow (PING Out)" -Direction Outbound -Protocol ICMPv4 -Action Allow -ErrorAction SilentlyContinue
    
    Write-Color "✓ Reglas ICMP aplicadas" $GREEN
}

# =========================
# FUNCIONES DE IP VIRTUAL (Windows no soporta IP virtuales igual que Linux)
# =========================

function Crear-IPVirtual {
    param([string]$ip, [string]$interfaz)
    
    Write-Color "--- VERIFICANDO IP VIRTUAL: $ip ---" $CYAN
    
    # En Windows, las IPs virtuales son IPs adicionales en la misma interfaz
    $existingIP = Get-NetIPAddress -InterfaceAlias $interfaz -IPAddress $ip -ErrorAction SilentlyContinue
    
    if ($existingIP) {
        Write-Color "✓ La IP $ip ya está configurada en $interfaz" $GREEN
        return $true
    }
    
    $ipPrincipal = Obtener-IPActual
    if ($ip -eq $ipPrincipal) {
        Write-Color "✗ Error: No puedes usar la IP principal ($ipPrincipal) como virtual" $RED
        return $false
    }
    
    Write-Color "➜ Creando IP virtual: $ip en $interfaz..." $YELLOW
    
    try {
        New-NetIPAddress -InterfaceAlias $interfaz -IPAddress $ip -PrefixLength 24 -ErrorAction Stop
        Write-Color "✓ IP virtual $ip creada exitosamente" $GREEN
        
        # Hacer persistente (agregar al registro)
        $regPath = "HKLM:\SYSTEM\CurrentControlSet\Services\Tcpip\Parameters\Interfaces\$((Get-NetAdapter -Name $interfaz).InterfaceGuid)"
        $currentIPs = Get-ItemProperty -Path $regPath -Name "IPAddress" -ErrorAction SilentlyContinue
        
        if ($currentIPs) {
            $newIPs = $currentIPs.IPAddress + $ip
            Set-ItemProperty -Path $regPath -Name "IPAddress" -Value $newIPs
        }
        
        $global:IPS_VIRTUALES += $ip
        
        Write-Color "`nEstado actual de $interfaz:" $CYAN
        Get-NetIPAddress -InterfaceAlias $interfaz | Where-Object { $_.AddressFamily -eq "IPv4" } | Format-Table IPAddress, PrefixLength
        
        return $true
    }
    catch {
        Write-Color "✗ Error al crear IP virtual: $_" $RED
        return $false
    }
}

function Eliminar-IPVirtual {
    param([string]$ip, [string]$interfaz)
    
    Write-Color "--- ELIMINANDO IP VIRTUAL: $ip ---" $CYAN
    
    $existingIP = Get-NetIPAddress -InterfaceAlias $interfaz -IPAddress $ip -ErrorAction SilentlyContinue
    
    if ($existingIP) {
        Write-Color "➜ Eliminando IP virtual: $ip de $interfaz..." $YELLOW
        
        try {
            Remove-NetIPAddress -InterfaceAlias $interfaz -IPAddress $ip -Confirm:$false -ErrorAction Stop
            Write-Color "✓ IP virtual $ip eliminada" $GREEN
            
            # Eliminar del registro
            $regPath = "HKLM:\SYSTEM\CurrentControlSet\Services\Tcpip\Parameters\Interfaces\$((Get-NetAdapter -Name $interfaz).InterfaceGuid)"
            $currentIPs = Get-ItemProperty -Path $regPath -Name "IPAddress" -ErrorAction SilentlyContinue
            
            if ($currentIPs) {
                $newIPs = $currentIPs.IPAddress | Where-Object { $_ -ne $ip }
                Set-ItemProperty -Path $regPath -Name "IPAddress" -Value $newIPs
            }
            
            Write-Color "`nEstado actual de $interfaz:" $CYAN
            Get-NetIPAddress -InterfaceAlias $interfaz | Where-Object { $_.AddressFamily -eq "IPv4" } | Format-Table IPAddress, PrefixLength
            
            return $true
        }
        catch {
            Write-Color "✗ Error al eliminar IP virtual: $_" $RED
            return $false
        }
    }
    else {
        Write-Color "⚠ La IP $ip no existe en $interfaz" $YELLOW
        return $true
    }
}
