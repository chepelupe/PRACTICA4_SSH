function Instalar-Todo {
    Write-Host ""
    Write-Host "--- INSTALANDO TODOS LOS ROLES ---"
    
    if (Get-Command "Instalar-DHCP" -ErrorAction SilentlyContinue) {
        Instalar-DHCP
    } else {
        Write-Host "ERROR: FUNCION Instalar-DHCP NO DISPONIBLE"
    }
    
    if (Get-Command "Instalar-DNS" -ErrorAction SilentlyContinue) {
        Instalar-DNS
    } else {
        Write-Host "ERROR: FUNCION Instalar-DNS NO DISPONIBLE"
    }
    
    if (Get-Command "Instalar-SSH" -ErrorAction SilentlyContinue) {
        Instalar-SSH
    } else {
        Write-Host "ERROR: FUNCION Instalar-SSH NO DISPONIBLE"
    }
    
    if (Get-Command "Configurar-FirewallPing" -ErrorAction SilentlyContinue) {
        Configurar-FirewallPing
    }
    
    Write-Host ""
    Write-Host "INSTALACION DE TODOS LOS ROLES COMPLETADA"
}

function Configurar-IPEstatica {
    if (-not $INTERFACE) {
        Write-Host ""
        Write-Host "PRIMERO DEBES SELECCIONAR UNA INTERFAZ"
        Seleccionar-Interfaz
    }
    
    $CURRENT_IP = Obtener-IPActual
    Write-Host ""
    Write-Host "--- CONFIGURACION IP ESTATICA ($INTERFACE) ---"
    Write-Host "IP ACTUAL: $(if ($CURRENT_IP) { $CURRENT_IP } else { 'NINGUNA' })"
    
    $resp = Read-Host "CONFIGURAR IP ESTATICA NUEVA? (S/N)"
    if ($resp -eq "s") {
        do {
            $nueva_ip = Read-Host "INGRESE IP ESTATICA"
            $ip_valida = Validar-IP $nueva_ip
            if (-not $ip_valida) {
                Write-Host "IP INVALIDA O PROHIBIDA."
            }
        } while (-not $ip_valida)

        Write-Host "APLICANDO CONFIGURACION..."
        
        # ELIMINAR IP EXISTENTE
        $existingIP = Get-NetIPAddress -InterfaceAlias $INTERFACE -AddressFamily IPv4 -ErrorAction SilentlyContinue
        if ($existingIP) {
            Remove-NetIPAddress -InterfaceAlias $INTERFACE -Confirm:$false -ErrorAction SilentlyContinue
        }
        
        # AGREGAR NUEVA IP
        try {
            New-NetIPAddress -InterfaceAlias $INTERFACE -IPAddress $nueva_ip -PrefixLength 24 -ErrorAction Stop
            Write-Host "IP $nueva_ip ASIGNADA A $INTERFACE"
        } catch {
            Write-Host "ERROR AL CONFIGURAR IP: $_"
        }
        
        Write-Host ""
        Write-Host "NOTA: ESTA CONFIGURACION ES TEMPORAL"
        Write-Host "PARA HACERLA PERMANENTE, CONFIGURA EN EL PANEL DE RED"
    }
}

function Verificar-Todo {
    Write-Host ""
    Write-Host "--- VERIFICACION COMPLETA DEL SISTEMA ---"
    
    # VERIFICAR INTERFAZ
    Write-Host ""
    Write-Host "INTERFAZ PRINCIPAL: $(if ($INTERFACE) { $INTERFACE } else { 'NO DEFINIDA' })"
    
    # VERIFICAR IPS
    Write-Host ""
    Write-Host "TODAS LAS INTERFACES DE RED:"
    Get-NetIPAddress -AddressFamily IPv4 | Where-Object { $_.InterfaceAlias -ne "Loopback" } | ForEach-Object {
        Write-Host "  $($_.InterfaceAlias): $($_.IPAddress)"
    }
    
    # VERIFICAR SERVICIOS
    Write-Host ""
    Write-Host "ESTADO DE SERVICIOS:"
    
    if (Get-Command "Verificar-DHCP" -ErrorAction SilentlyContinue) {
        Verificar-DHCP
    }
    if (Get-Command "Verificar-DNS" -ErrorAction SilentlyContinue) {
        Verificar-DNS
    }
    if (Get-Command "Verificar-SSH" -ErrorAction SilentlyContinue) {
        Verificar-SSH
    }
}

function Info-Clientes {
    $SERVER_IP = Obtener-IPActual
    Write-Host ""
    Write-Host "--- INFORMACION PARA CLIENTES ---"
    Write-Host ""
    Write-Host "IP DEL SERVIDOR: $SERVER_IP"
    Write-Host "USUARIO: $env:USERNAME"
    Write-Host ""
    Write-Host "COMANDOS DE CONEXION:"
    Write-Host "  SSH: ssh $env:USERNAME@$SERVER_IP"
    Write-Host "  RDP: mstsc /v:$SERVER_IP"
    Write-Host ""
    
    if ($global:SSH_IP_EXCLUSIVA) {
        Write-Host "PUERTO EXCLUSIVO SSH:"
        Write-Host "  ssh $env:USERNAME@$global:SSH_IP_EXCLUSIVA"
    }
}
