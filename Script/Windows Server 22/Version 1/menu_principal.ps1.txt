function Instalar-TodosRoles {

}

function Configurar-IPEstatica {
    $CURRENT_IP = Obtener-IPActual
    Write-Color "`n--- CONFIGURACIÓN IP ESTÁTICA ($INTERFACE) ---" $YELLOW
    Write-Host "IP Actual: $CURRENT_IP"
    
    $resp = Read-Host "`n¿Configurar IP ESTATICA nueva? (s/n)"
    if ($resp -eq "s") {
        do {
            $nuevaIP = Read-Host "Ingrese IP Estática"
        } while (-not (Validar-IPCompleta $nuevaIP))
        
        $mask = Read-Host "Máscara de red [Enter=24]" 
        if ([string]::IsNullOrEmpty($mask)) { $mask = 24 }
        
        $gateway = Read-Host "Gateway (Enter para omitir)"
        
        Write-Color "Aplicando configuración..." $CYAN
        
        try {
            # Eliminar IP actual
            Get-NetIPAddress -InterfaceAlias $INTERFACE -AddressFamily IPv4 | Remove-NetIPAddress -Confirm:$false
            
            # Configurar nueva IP
            if ([string]::IsNullOrEmpty($gateway)) {
                New-NetIPAddress -InterfaceAlias $INTERFACE -IPAddress $nuevaIP -PrefixLength $mask
            } else {
                New-NetIPAddress -InterfaceAlias $INTERFACE -IPAddress $nuevaIP -PrefixLength $mask -DefaultGateway $gateway
            }
            
            # Configurar DNS
            Set-DnsClientServerAddress -InterfaceAlias $INTERFACE -ServerAddresses "127.0.0.1"
            
            Configurar-FirewallPing
            
            Write-Color "IP $nuevaIP asignada a $INTERFACE." $GREEN
        }
        catch {
            Write-Color "Error al configurar IP: $_" $RED
        }
    }
}

function Verificar-InstalacionCompleta {
    Write-Color "`n--- VERIFICACIÓN COMPLETA DEL SISTEMA ---" $CYAN
    
    # Verificar IPs
    Write-Color "`nIPs configuradas en $INTERFACE:" $CYAN
    Get-NetIPAddress -InterfaceAlias $INTERFACE -AddressFamily IPv4 | Format-Table IPAddress, PrefixLength
    
    # Verificar servicios
    Write-Color "`nEstado de servicios:" $CYAN
    
    # DHCP
    $dhcpServer = Get-WindowsFeature -Name DHCP
    if ($dhcpServer.Installed) {
        Write-Host "  DHCP: Instalado" -ForegroundColor Green
        $dhcpService = Get-Service -Name DHCPServer -ErrorAction SilentlyContinue
        if ($dhcpService -and $dhcpService.Status -eq "Running") {
            Write-Host "         ✓ Activo" -ForegroundColor Green
        } else {
            Write-Host "         ✗ Inactivo" -ForegroundColor Red
        }
    }
    
    # DNS
    $dnsServer = Get-WindowsFeature -Name DNS
    if ($dnsServer.Installed) {
        Write-Host "  DNS:   Instalado" -ForegroundColor Green
        $dnsService = Get-Service -Name DNS -ErrorAction SilentlyContinue
        if ($dnsService -and $dnsService.Status -eq "Running") {
            Write-Host "         ✓ Activo" -ForegroundColor Green
        } else {
            Write-Host "         ✗ Inactivo" -ForegroundColor Red
        }
    }
    
    # SSH
    $sshCapability = Get-WindowsCapability -Online | Where-Object { $_.Name -like 'OpenSSH.Server*' }
    if ($sshCapability.State -eq 'Installed') {
        Write-Host "  SSH:   Instalado" -ForegroundColor Green
        $sshService = Get-Service -Name sshd -ErrorAction SilentlyContinue
        if ($sshService -and $sshService.Status -eq "Running") {
            Write-Host "         ✓ Activo" -ForegroundColor Green
        } else {
            Write-Host "         ✗ Inactivo" -ForegroundColor Red
        }
    }
}

function Generar-InformacionConexion {
    $SERVER_IP = Obtener-IPActual
    
    Write-Color "`n--- INFORMACIÓN DE CONEXIÓN PARA CLIENTES ---" $CYAN
    Write-Color "`nServidor Windows Server 2022:" $YELLOW
    Write-Host "  IP: $SERVER_IP"
    Write-Host "  Usuario: $env:USERNAME"
    Write-Host "  Dominio: $env:USERDOMAIN"
    
    Write-Color "`nPara conectarte por SSH:" $GREEN
    Write-Host "  ssh $env:USERNAME@$SERVER_IP"
    
    Write-Color "`nPara configurar DNS en clientes:" $GREEN
    Write-Host "  En Windows: netsh interface ip set dns name='Ethernet' static $SERVER_IP"
    Write-Host "  En Linux: echo 'nameserver $SERVER_IP' > /etc/resolv.conf"
    
    Read-Host "`nPresiona Enter para continuar..."
}
